- name: deploymeny
  hosts: all
  become: true
  vars_files:
    - vars_2.yaml

  tasks:
    - name: install {{ packages }}
      yum:
        name: "{{ packages }}"
        state: present
    - name: Create a login user
      user:
        name: "{{ user }}"
        password: "{{ password | password_hash('sha512') }}"
        state: present
        shell: /bin/bash
        system: yes
        createhome: yes
        home: /home/"{{ user }}"
    - name: Create a file
      file:
        path: "{{ create_file }}"
        state: directory
        owner: "{{ user }}"
        group: "{{ user }}"
        mode: "0644"
    - name : copy files to /opt/backup
      copy:
        src: "{{ item }}"
        dest: "{{ create_file }}"
        remote_src: yes
      with_items: "{{ files }}"
